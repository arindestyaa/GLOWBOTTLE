    <script>
        // Mengambil elemen form dan pesan status
        const form = document.getElementById('dicosma-form');
        const statusMessage = document.getElementById('status-message');

        // Fungsi untuk menampilkan pesan status
        function showStatus(message, isSuccess = true) {
            statusMessage.textContent = message;
            statusMessage.classList.remove('hidden', 'bg-red-50', 'text-red-700', 'border-red-200');
            statusMessage.classList.remove('bg-green-50', 'text-green-700', 'border-green-200');
            
            if (isSuccess) {
                statusMessage.classList.add('bg-green-50', 'text-green-700', 'border-green-200');
            } else {
                statusMessage.classList.add('bg-red-50', 'text-red-700', 'border-red-200');
            }

            // Otomatis sembunyikan setelah 5 detik
            setTimeout(() => {
                statusMessage.classList.add('hidden');
            }, 5000);
        }

        // Event listener untuk pengiriman form
        form.addEventListener('submit', function(event) {
            // Mencegah pengiriman form default
            event.preventDefault();

            // Mengumpulkan data dari form
            const dataPendaftar = {
                nama_lengkap: document.getElementById('nama_lengkap').value,
                kelas: document.getElementById('kelas').value,
                no_whatsapp: document.getElementById('no_whatsapp').value,
                alasan_mendaftar: document.getElementById('alasan_mendaftar').value,
                organisasi: 'BLINK', // Menambahkan identitas organisasi
                tanggal_daftar: new Date().toISOString()
            };

            // Log data ke konsol (Simulasi penyimpanan data. Dalam aplikasi nyata, data ini akan dikirim ke server/Firestore.)
            console.log('--- Data Pendaftaran Anggota BLINK Baru ---');
            console.log(dataPendaftar);
            console.log('-------------------------------------------');

            // Tampilkan pesan sukses kepada pengguna
            showStatus('Pendaftaran untuk BLINK berhasil! Data Anda telah kami terima. Silakan cek konsol untuk melihat data yang disimulasikan.', true);

            // Bersihkan form
            form.reset();
        });
    </script>
